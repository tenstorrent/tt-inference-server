# Dependencies for the V2_PREFIX_CACHE workflow venv (Python 3.11).
# Used by direct invocations of tt-inference-server-v2/run.py when
# --prefix-cache is set.
#
# This is a strict superset of requirements/benchmarks-aiperf.txt: the
# v2 orchestrator additionally imports aiohttp at module-load time via
# workflow_module -> test_module.context -> health_tests. Every other
# v2 test runner (audio / image / video / CNN / TTS / embedding evals
# and benchmarks, plus run_llm_performance) is lazy-loaded via PEP 562
# __getattr__ in test_module/__init__.py and
# test_module/llm_tests/__init__.py and is NEVER imported on the
# prefix-cache code path -- so their transitive deps (torchvision,
# open-clip-torch, pillow, scipy, numpy) intentionally do NOT appear
# below. pip will still pull in numpy via transformers / aiperf, but
# we leave that to pip's resolver instead of pinning it ourselves.

-c constraints.txt
--extra-index-url https://download.pytorch.org/whl/cpu

# AIPerf benchmarking tool (https://github.com/ai-dynamo/aiperf).
# transformers>=5 is required to load newer tokenizers (e.g. Gemma 4, whose
# tokenizer_config.json stores `extra_special_tokens` as a list that only
# transformers 5.x parses; 4.x raises `AttributeError: 'list' object has no
# attribute 'keys'`). transformers 5.x imports `torch.distributed.tensor.
# device_mesh` on `AutoTokenizer`, which only exists in torch>=2.5, so the
# CPU torch pin is bumped to 2.5.1. aiperf itself only requires transformers>=4.56.
torch==2.5.1+cpu
aiperf
transformers>=5.10.2,<6
# Gemma (and other SentencePiece) tokenizers need protobuf; sentencepiece is
# its usual companion for the slow-tokenizer conversion path.
protobuf
sentencepiece
pyjwt
requests
datasets

# v2 orchestrator eager-import: workflow_module -> test_module.context ->
# health_tests uses aiohttp at module top.
aiohttp
